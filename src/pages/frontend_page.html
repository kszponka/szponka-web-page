@mixin / @include

                
@mixin borderBackground {
   border-radius: 5px; 
   background: blue; 
  }

.main {
  @include borderBackground; 
} 

------ SCSS => CSS ------

.main {
  border-radius: 5px; 
  background: blue; 
}

params

                
@mixin borderBackgroundParams($radius, $background) {
  border-radius: $radius; 
  background: $background; 
} 

.main {
  @include borderBackgroundParams(5px, pink); 
} 

------ SCSS => CSS ------

.main {
  border-radius: 5px;
  background: pink; 
}
